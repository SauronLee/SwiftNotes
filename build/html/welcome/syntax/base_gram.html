    1let age = 4
2if age >= 22{
3    print("Get married")

7    print("Just a child")

    While¶

    1var num = 5
2while num > 0 {
3    print("num is \(num)")
4    num -= 1
5} // for five times

    For¶

    閉区間演算子 a...b means a <= range <= b

    1let names = ["A","B","C","D"]
2for i in 0...3 {
3    print(names[i])
4} // output: A,B,C,D

    1let range = 1...3
2for i in range {
3    print(names[i])
4} // output: B,C,D

    1for var i in range {
2    i += 5
3    print(i)
4} //output: 6,7,8

    半開放区間演算子 a..<b means a <= range < b

    1for i in 1..<5 {
2    print(i)
3} // output: 1,2,3,4

    1var functions: [() -> Int] = []
2for i in 1...3 {
3    functions.append {i}
4}
5for f in functions {
6    print(f())
7}
8// 1 2 3

    Switch with tuple¶

     1let point = (1, 1)
 2switch point {
 3    case (0, 0):
 4        print("the origin")
 5    case (_, 0):
 6        print("on the x-axis")
 7    case (0, _):
 8        print("on the y-axis")
 9    case (-2...2, -2...2):
10        print("inside the box")
11    default:
12        print("outside of the box")
13}   // inside the box

    1let point = (2, 0)
2switch point {
3    case (let x, 0):
4        print("on the x-axis with an x value of \(x)")
5    case (0, let y):
6        print("on the y-axis with an y value of \(y)")
7    case let (x, y):
8        print("somewhere else at (\(x), \(y))")
9}   // output: on the x-axis with an x value of 2

7    case let (x, y):

    Function Overload¶

     1func sum(v1: Int, v2: Int) -> Int {
 2    v1 + v2
 3}
 4func sum(v1: Int, v2: Int, v3: Int) -> Int {
 5    v1 + v2 + v3
 6}
 7func sum(v1: Int, v2: Double) -> Double {
 8    Double(v1) + v2
 9}
10func sum(v1: Double, v2: Int) -> Double {
11    v1 + Double(v2)
12}
13func sum(_ v1: Int, _ v2: Int) -> Int {
14    v1 + v2
15}
16func sum(a: Int, b: Int) -> Int {
17    a + b
18}
19sum(v1: 10, v2: 20) // 30
20sum(v1: 10, v2: 20, v3: 30) // 60
21sum(v1: 10, v2: 20.0) // 30.0
22sum(v1: 10.0, v2: 20) // 30.0
23sum(10, 20) // 30
24sum(a: 10, b: 20) // 30
