    Stored Property¶

    The essence is a member variable stored in instance memory |check| Struct (need initial value) |check| Class (need initial value) |uncheck| Enumerate

    1struct point {
2    var x: Int

    Computed Property¶

    The essence is the function
    Not stored in instance memory
    Only var can be used

    |check| Struct |check| Class |check| Enumerate

 2    var radius: Double

 4        set(newDiameter) {
 5            radius = newDiameter / 2 //newValue
 6        }

 9        }
10    }
11}

    Lazy Stored property

     1class Car{
 2    init() {
 3        print("Car init!")
 4    }
 5    func run() {
 6        print("Car is running!")
 7    }
 8}
 9class Person {
10    lazy var car = Car()
11    init() {
12        print("Person init!")
13    }
14    func goOut() {
15        car.run()
16    }
17}
18// test
19let p = Person()
20// output: Person init!
21p.goOut()
22// output:
23// Car init!
24// Car is running!

    Property Observer¶

     1struct Circle {
 2    var radius: Double {
 3        willSet {
 4            print("willSet", newValue)
 5        }
 6        didSet {
 7            print("didSet", oldValue, radius)
 8        }
 9    }
10    init() {
11        self.radius = 1.0
12        print("Circle init!")
13    }
14}
15// test
16var circle = Circle() // Circle init!
17circle.radius = 10.5 // willSet 10.5 \n didSet 1.0 10.5
18print(circle.radius) // 10.5

    Inout¶

    The essence of inout is address passing.

     1struct Shape {
 2    var width: Int
 3    var side: Int {
 4        willSet {
 5            print("willSetSide", newValue)
 6        }
 7        didSet {
 8            print("didSetSide", oldValue, side)
 9        }
10    }
11    var girth: Int {
12        set {
13            width = newValue / side
14            print("setGirth", newValue)
15        }
16        get {
17            print("getGirth")
18            return width * side
19        }
20    }
21    func show() {
22        print("width=\(width), side=\(side), girth=\(girth)")
23    }
24}
25// test
26func test(_ num: inout Int) {
27    num = 20
28}
29var s = Shape(width: 10, side: 4)
30test(&s.width)
31s.show()
32print("---")
33test(&s.side)
34s.show()
35print("---")
36test(&s.girth)
37s.show()
38/*
39getGirth
40width=20, side=4, girth=80
41---
42willSetSide 20
43didSetSide 4 20
44getGirth
45width=20, side=20, girth=400
46---
47getGirth
48setGirth 20
49getGirth
50width=1, side=20, girth=20
51*/
