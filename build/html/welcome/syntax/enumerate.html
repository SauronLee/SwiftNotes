    Recursive Enumeration¶

     1indirect enum ArithExpr {
 2    case number(Int)
 3    case sum(ArithExpr, ArithExpr)
 4    case difference(ArithExpr, ArithExpr)
 5}
 6// or
 7enum ArithExpr {
 8    case number(Int)
 9    indirect case sum(ArithExpr, ArithExpr)
10    indirect case difference(ArithExpr, ArithExpr)
11}

     1let five = ArithExpr.number(5)
 2let four = ArithExpr.number(4)
 3let two = ArithExpr.number(2)
 4let sum = ArithExpr.sum(five, four)
 5let difference = ArithExpr.difference(sum, two)
 6
 7func calculate(_ expr: ArithExpr) -> Int {
 8    switch expr {
 9        case let .number(value):
10            return value
11        case let .sum(left, right):
12            return calculate(left) + calculate(right)
13        case let .difference(left, right):
14            return calculate(left) - calculate(right)
15    }
16}
